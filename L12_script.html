2.2 模型1：年龄的立方 + sex

# lrm()函数
logit_rcs <- lrm(status~rcs(age,4) + sex,
                 data=df)
# 非线性显著性检验
anova(logit_rcs)
                Wald Statistics          Response: status 

 Factor     Chi-Square d.f. P     
 age        21.99      3    0.0001
  Nonlinear  8.73      2    0.0127
 sex         2.10      1    0.1473
 TOTAL      22.67      4    0.0001
# 保存估计结果，注意查看logit_rcs_pred的内容
# np = 100, 定量变量等分100等份；定性变量取最后一个水平
logit_rcs_pred <- Predict(logit_rcs, 
                          age, 
                          sex,
                          np = 100, 
                          fun = exp) 

logit_rcs_pred %>% 
  ggplot(aes(age, yhat, col = sex)) +
  geom_point() +
  scale_color_manual(values = c("green", "blue")) +
  geom_hline(yintercept = 1, linetype = 2) +
  geom_vline(xintercept = 50, linetype = 2) +
  geom_vline(xintercept = 45, linetype = 2) +
  theme_bw() +
  theme(legend.position = "bottom") +
  labs(title = "RCS of Age",
       x = "Age",
       y = "Odds Ratio")

3 Cox回归

3.1 模型1:年龄的立方 + 性别

library(survival)
#Surve()函数中要求outcome是0和1
df$status <- ifelse(df$status == "Dead", 1, 0)

dd<-datadist(df)
options(datadist='dd')

cox_rcs <- cph(Surv(time, status) ~ rcs(age,4) + sex,
               data = df)

anova(cox_rcs)
                Wald Statistics          Response: Surv(time, status) 

 Factor     Chi-Square d.f. P     
 age        32.99      3    <.0001
  Nonlinear 10.08      2    0.0065
 sex         3.07      1    0.0796
 TOTAL      34.50      4    <.0001
cox_rcs_pred_HR <- Predict(cox_rcs, 
                          age,
                          np = 100, 
                          fun = exp,
                          ref.zero = TRUE) 

cox_rcs_pred_HR %>% 
  ggplot(aes(age, yhat)) +
  geom_point(col = "cyan")  +
  geom_hline(yintercept = 1, linetype = 2) +
  theme_bw() +
  theme(legend.position = "bottom") +
  labs(title = "RCS of Age & Size",
       x = "Age",
       y = "HR")

# 绘制分性别曲线
cox_rcs_pred_HR <- Predict(cox_rcs, 
                           size,
                           age = quantile(df$age, seq(0.1, 0.9, 0.2)),
                           sex = c("Female", "Male"),
                           np = 100,
                           fun = exp,
                           ref.zero = TRUE) %>% 
  data.frame()

cox_rcs_pred_HR %>% 
  ggplot(aes(size, yhat, col = sex)) +
  geom_point() +
  facet_wrap( ~ age) +
  geom_ribbon(aes(ymin = lower, ymax = upper), alpha = 0.2) +
  geom_hline(yintercept = 1, linetype = 3) +
  theme_bw() +
  theme(legend.position = "bottom") +
  labs(title = "RCS of Size",
       x = "Size",
      y = "HR")
