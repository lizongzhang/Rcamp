4 DAG绘制

library(ggdag)
library(tidyverse)

# Create super basic DAG
simple_dag <- dagify(
  FEV ~ SMOKE,
  FEV ~ AGE,
  SMOKE ~ AGE,
  exposure = "SMOKE",
  outcome = "FEV"
)

# ggdag() 绘制DAG
ggdag(simple_dag) +
  theme_dag() #添加主题

#ggdag_status() 绘制标识暴露和结局的DAG 更常用
ggdag_status(simple_dag) + #单独运行ggdag_status()查看坐标系
  theme_dag_blank()

simple_dag_with_coords <- dagify(
  FEV ~ SMOKE,
  FEV ~ AGE,
  SMOKE ~ AGE,
  exposure = "SMOKE",
  outcome = "FEV",
  labels = c(FEV = "FEV",
             SMOKE = "Smoking",
             AGE = "Age"),
  coords = list(x = c(FEV = 3, SMOKE = -3, AGE = 0),  #自定义节点坐标
                y = c(FEV = 0, SMOKE = 0, AGE = 0.5))
)

# 自定义节点大小
ggdag_status(simple_dag_with_coords,
             node_size = 22) +
  guides(fill = FALSE, color = FALSE) +  # 不显示图例
  theme_dag_gray_grid()

df$AGE <- as.numeric(df$AGE)
df$SEX <- as.numeric(df$SEX) - 1  # sex = 1, Male; sex = 0, Female;
