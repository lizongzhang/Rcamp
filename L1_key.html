1.10 用hist()函数画出airquality数据框中的Temp的直方图。将改图片保存在项目路径下。

hist(airquality$Temp,
     breaks = c(50, 60, 70, 80, 90, 100),
     col = "cyan4",
     border = "black",
     xlab = "Temperature (degrees F)",
     ylab = "Frequency",
     main = "Temperature (degrees F) in New York City, May to September 1973",
     xlim = c(50, 100),
     ylim = c(0, 60)
     )

# 获取直方图的数据
hist_data <- hist(airquality$Temp,
                  breaks = c(50, 60, 70, 80, 90, 100),
                  plot = FALSE)

# 在直方图上标注频数
text(x = hist_data$mids, y = hist_data$counts, 
     labels = hist_data$counts, pos = 3, col = 1)
