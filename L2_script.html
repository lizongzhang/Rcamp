3.3 图形类

3.3.1 直方图hist()

hist(mtcars$mpg)

### 创建直方图，组距为5
hist(mtcars$mpg, 
     breaks = seq(10,35,5),                   # 组距为5
     col = 6,                                 # 填充色
     border = 1,                              # 边框色
     lty = 1,                                 # 边框线条形状
     lwd = 1.2,                               # 线条宽度，默认值的1.2倍
     main = "Histogram of Miles per Gallon",  # 添加图形标题
     xlab = "Miles per Gallon",               # 添加X轴标签
     ylab = "Frequency",                      # 添加Y轴标签
     xlim = c(10, 35),                        # 设置X轴刻度范围
     ylim = c(0, 14),                         # 设置Y轴刻度范围
     las = 1,                                 #旋转Y轴数字至水平方向
     cex.main = 1.5,                           #图形标题是默认值的1.5倍
     cex.lab = 1.2,                           #坐标轴标签字体是默认值的1.2倍
     cex.axis = 1.2,                          #坐标轴刻度字体是默认值的1.2倍
)

# 获取直方图的数据
hist_data <- hist(mtcars$mpg, 
                  breaks = seq(10,35,5),

# 在直方图上标注频数
text(x = hist_data$mids, y = hist_data$counts,

3.3.3 条形图barplot()

# 创建一个数据表，统计gear的分布
counts <- table(mtcars$gear)

# 使用barplot()函数创建条形图
barplot(counts,               # 条形图的高度数据，即每个齿轮数量的次数
        col = 3,              # 指定条形的颜色，这里使用数字2代表红色
        main="Car Distribution",  # 指定图表的标题
        xlab="Number of Gears",   # 指定x轴标签
        ylim = c(0, 18),        # 指定y轴的数值范围，从0到18
        yaxt = "n")             # 设置Y轴刻度的类型为 "n"，表示不显示Y轴刻度

# 手动设置Y轴刻度
axis(2, at = seq(0, 18, by = 3))

# 在每个条形上标注频数
text(x = barplot(counts, plot = FALSE), 
     y = counts + 0.1,            # 将文本标签略微上移，以避免与条形重叠
     labels = counts, 
     pos = 3,                   # 文本标签的位置，3表示在条形的上方
     col = "blue")              # 文本标签的颜色

# Simple Horizontal Bar Plot with Added Labels
counts <- table(mtcars$gear)
barplot(counts, 
        horiz =TRUE,
        main="Car Distribution", 
        names.arg=c("3 Gears", "4 Gears", "5 Gears"),
        las = 1,
        xlim = c(0,18),
        xaxt = "n")

# 手动设置Y轴刻度
axis(1, at = seq(0, 18, by = 3))
