2.8 分组生存曲线

# 创建分性别预测个案集
sex_df <- data.frame(sex = c(1,2),
                     age = rep(mean(lung$age, na.rm = TRUE), 2),
                     ph.ecog = c(1, 1))

# 分性别生存曲线 Survival curves
fit <- survfit(res.cox, newdata = sex_df)
ggsurvplot(fit, conf.int = TRUE, 
           data = lung, 
           risk.table = TRUE,
           legend.labs=c("Sex = Male", "Sex = Female"),
           ggtheme = theme_minimal())

# 分年龄段survival rates
lung$age3 <- cut(lung$age,
                 c(35,55,65,85))
table(lung$age3)

(35,55] (55,65] (65,85] 
     49      87      92 
fit <- coxph(Surv(time, status) ~
             sex + ph.ecog + age +strata(age3),
            data = lung)

survminer::ggadjustedcurves(fit, 
                 data = lung,
                 variable = "age3")
